"""Limit of detection (LOD) calculator for heavy metal ions.

Connects LSPR wavelength shift to ion concentration via effective surface
dn/dc values for thiol-functionalized AuNP LSPR sensors.

Physical basis for dn/dc values:
  In functionalized LSPR sensors, heavy metal ions bind specifically to
  surface-immobilized thiol ligands (not bulk RI change). The effective
  dn/dc accounts for:
    (a) the Langmuir surface concentration enhancement (Ka * Gamma_max)
    (b) the local RI change per bound metal-thiol complex
    (c) the evanescent field sampling fraction (t_layer / l_d)

  These values are back-calculated from verified experimental LSPR
  calibration data for analogous AuNP-fiber sensors, assuming our
  instrument resolution sigma = 0.01 nm and a reference S_LSPR ~ 80 nm/RIU.
  Uncertainty is estimated at +/-50% due to ligand-system differences.

References (verified):
  Cd2+: Solomonea et al. (2022) Biosensors 12(8):573
  Pb2+: Boruah & Biswas (2018) Sens. Actuators B 276:89-94
  Hg2+: Sadani et al. (2019) Biosens. Bioelectron. 134:90-96
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from .plot_style import set_journal_style, save_figure, FIGURE_SIZES, ION_COLORS, ION_MARKERS


# Instrument wavelength resolution (Ocean Insight USB4000 fiber spectrometer)
_SIGMA_LAMBDA_NM = 0.01   # nm


def get_ion_parameters():
    """Return physical/chemical parameters for the three target heavy metal ions.

    dndc_RIU_per_molL values are SURFACE-EFFECTIVE dn/dc values for
    thiol-functionalized AuNP LSPR sensors, NOT bulk solution dn/dc.
    They account for ligand-mediated surface concentration enhancement
    (Ka x Gamma_max ~ 10^4-10^6) relative to bulk solution.

    Calibration sources (back-calculated at sigma=0.01 nm, S_ref=80 nm/RIU):
      Cd2+ (L-Cys): Solomonea et al. 2022 Biosensors 12(8):573
                    LOD=0.80 ug/L (7.1 nM), FO-SPR w/ BSA receptor:
                    dn_eff/dc ~ 53,000 RIU/(mol/L).
      Pb2+ (GSH):   Boruah & Biswas 2018 Sens. Actuators B 276:89-94
                    LOD=1.87 ug/L, U-bend LSPR w/ oxalic-acid AuNPs:
                    dn_eff/dc ~ 42,000 RIU/(mol/L).
      Hg2+ (11-MUA):Sadani et al. 2019 Biosens. Bioelectron. 134:90-96
                    LOD=0.10 ug/L, U-bend LSPR w/ chitosan-AuNPs on BSA
                    (Ka~10^9-10^10 M^-1 for Hg-thiol):
                    dn_eff/dc ~ 750,000 RIU/(mol/L).

    Returns
    -------
    dict
        Nested dict keyed by ion name.
    """
    return {
        'Cd2+': {
            'molar_mass':        112.4,     # g/mol
            'who_limit_ug_L':    3.0,
            'dndc_RIU_per_molL': 53000.0,  # effective surface dn/dc (Solomonea 2022)
            'ligand':            'L-Cysteine (thiol)',
        },
        'Pb2+': {
            'molar_mass':        207.2,
            'who_limit_ug_L':    10.0,
            'dndc_RIU_per_molL': 42000.0,  # effective surface dn/dc (Boruah 2018)
            'ligand':            'Glutathione (thiol)',
        },
        'Hg2+': {
            'molar_mass':        200.6,
            'who_limit_ug_L':    1.0,
            'dndc_RIU_per_molL': 750000.0, # effective surface dn/dc (Sadani 2019)
            'ligand':            '11-MUA (thiol)',
        },
    }


def compute_lod(ion_name, S_total_nm_per_RIU, sigma_lambda_nm=_SIGMA_LAMBDA_NM):
    """Compute limit of detection for a single ion.

    LOD = 3σ / S_total  (3-sigma criterion, standard in analytical chemistry)

    Steps:
      a) Δn at LOD = 3σ / S_total
      b) c [mol/L] = Δn / (dn/dc)
      c) c [µg/L]  = c [mol/L] × M [g/mol] × 10^6

    Parameters
    ----------
    ion_name : str
        One of 'Cd2+', 'Pb2+', 'Hg2+'.
    S_total_nm_per_RIU : float
        Composite sensitivity in nm/RIU.
    sigma_lambda_nm : float
        Instrument wavelength resolution (default 0.01 nm).

    Returns
    -------
    dict
        Keys: 'LOD_ug_L', 'LOD_uM', 'LOD_mol_L', 'delta_n_at_LOD',
              'delta_lambda_at_LOD_nm', 'exceeds_who_limit'
    """
    params = get_ion_parameters()[ion_name]
    delta_lambda = 3 * sigma_lambda_nm                          # nm
    delta_n      = delta_lambda / S_total_nm_per_RIU           # RIU
    c_mol_L      = delta_n / params['dndc_RIU_per_molL']       # mol/L
    c_g_L        = c_mol_L * params['molar_mass']              # g/L
    c_ug_L       = c_g_L * 1e6                                  # µg/L
    c_uM         = c_mol_L * 1e6                                # µM

    return {
        'LOD_ug_L':              float(c_ug_L),
        'LOD_uM':                float(c_uM),
        'LOD_mol_L':             float(c_mol_L),
        'delta_n_at_LOD':        float(delta_n),
        'delta_lambda_at_LOD_nm': float(delta_lambda),
        'exceeds_who_limit':     bool(c_ug_L > params['who_limit_ug_L']),
    }


def compute_all_lods(S_total_nm_per_RIU, sigma_lambda_nm=_SIGMA_LAMBDA_NM):
    """Compute LOD for all three ions.

    Parameters
    ----------
    S_total_nm_per_RIU : float
    sigma_lambda_nm : float

    Returns
    -------
    pandas.DataFrame
        Indexed by ion name.
    """
    ions = get_ion_parameters()
    rows = {}
    for ion in ions:
        result = compute_lod(ion, S_total_nm_per_RIU, sigma_lambda_nm)
        result['who_limit_ug_L'] = ions[ion]['who_limit_ug_L']
        result['ligand']         = ions[ion]['ligand']
        rows[ion] = result
    return pd.DataFrame(rows).T

def plot_lod_comparison(S_total_nm_per_RIU,
                         save_path='figures/fig6_lod_comparison'):
    """Two-panel LOD comparison figure.

    Left: bar chart with WHO thresholds.
    Right: benchmark scatter vs literature.

    Parameters
    ----------
    S_total_nm_per_RIU : float
    save_path : str
    """
    set_journal_style()
    lod_df   = compute_all_lods(S_total_nm_per_RIU)
    lit_df   = benchmark_against_literature()
    ions     = ['Cd2+', 'Pb2+', 'Hg2+']
    ion_pars = get_ion_parameters()

    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=FIGURE_SIZES['double_column'])

    # Left panel: horizontal bar chart
    lod_vals  = [lod_df.loc[ion, 'LOD_ug_L'] for ion in ions]
    who_vals  = [ion_pars[ion]['who_limit_ug_L'] for ion in ions]
    colors    = [ION_COLORS[ion] for ion in ions]
    y_pos     = np.arange(len(ions))

    ax1.barh(y_pos, lod_vals, color=colors, height=0.5, log=True)
    for i, (lod, who, ion) in enumerate(zip(lod_vals, who_vals, ions)):
        ax1.axvline(who, color=ION_COLORS[ion], ls='--', lw=1.2, alpha=0.7)
        ax1.text(who * 1.1, i, f'WHO: {who} µg/L', va='center', fontsize=7,
                 color=ION_COLORS[ion])
        ax1.text(lod * 1.1, i, f'{lod:.3f}', va='center', fontsize=7)

    ax1.set_yticks(y_pos)
    ax1.set_yticklabels(ions)
    ax1.set_xlabel('LOD (µg/L)', fontsize=10)
    ax1.set_xscale('log')
    ax1.set_title('Predicted LOD vs WHO limits')

    # Right panel: benchmark scatter
    method_labels = ['This simulation'] + lit_df['Reference'].tolist()
    x_ticks = np.arange(len(method_labels))

    for ion in ions:
        # This simulation
        ax2.scatter([0], [lod_df.loc[ion, 'LOD_ug_L']],
                    marker=ION_MARKERS[ion], color=ION_COLORS[ion],
                    s=80, zorder=5, label=ion)
        # Literature
        ion_lit = lit_df[lit_df['Ion'] == ion]
        for _, row in ion_lit.iterrows():
            ref_idx = method_labels.index(row['Reference'])
            ax2.scatter([ref_idx], [row['LOD_ug_L']],
                        marker=ION_MARKERS[ion], color=ION_COLORS[ion],
                        s=40, alpha=0.7)
        # WHO line
        who = ion_pars[ion]['who_limit_ug_L']
        ax2.axhline(who, color=ION_COLORS[ion], ls=':', lw=1, alpha=0.5)

    ax2.set_yscale('log')
    ax2.set_xticks(x_ticks)
    ax2.set_xticklabels(method_labels, rotation=35, ha='right', fontsize=7)
    ax2.set_ylabel('LOD (µg/L)', fontsize=10)
    ax2.set_title('Benchmark vs literature')
    ax2.legend(fontsize=8, title='Ion')

    save_figure(fig, save_path)
    plt.close(fig)


def plot_sensitivity_lod_tradeoff(save_path='figures/fig7_sensitivity_lod_tradeoff'):
    """Log-log plot of LOD vs composite sensitivity for each ion.

    Parameters
    ----------
    save_path : str
    """
    set_journal_style()
    S_range  = np.logspace(np.log10(50), np.log10(500), 200)
    ions     = ['Cd2+', 'Pb2+', 'Hg2+']
    ion_pars = get_ion_parameters()

    fig, ax = plt.subplots(figsize=FIGURE_SIZES['single_column_tall'])

    for ion in ions:
        lods = [compute_lod(ion, S)['LOD_ug_L'] for S in S_range]
        ax.loglog(S_range, lods, color=ION_COLORS[ion], lw=1.8, label=ion)
        who = ion_pars[ion]['who_limit_ug_L']
        ax.axhline(who, color=ION_COLORS[ion], ls='--', lw=0.9, alpha=0.6)
        ax.text(S_range[-1] * 0.95, who * 1.05, f'WHO ({ion})',
                ha='right', fontsize=7, color=ION_COLORS[ion])

    ax.set_xlabel('Composite sensitivity $S_{total}$ (nm/RIU)', fontsize=10)
    ax.set_ylabel('LOD (µg/L)', fontsize=10)
    ax.set_title('LOD vs sensor sensitivity', fontsize=11)
    ax.legend(fontsize=9)
    ax.invert_yaxis()   # lower LOD (better) at top

    save_figure(fig, save_path)
    plt.close(fig)
